Place the copied sign down:

$${
SPLIT(">-<",%@&signdata%,&sign[]);
PLACESIGN(%&sign[0]%,%&sign[1]%,%&sign[2]%,%&sign[3]%);
LOG(&6&lSign Pasted);
}$$


Copy sign that you are looking at:

$${
IF(HITNAME == "Sign");
	JOIN(">-<",SIGNTEXT[],@&signdata);
	STRIP(@&signdata,%@&signdata%);
	LOG(&6&lSign Copied);
ELSE;
	LOG(&4&lNo Sign Found)
ENDIF;
}$$
